const stillArrow = document.getElementById('stillUp'); stillArrow.addEventListener('click',()=>{ var stillWrapper = document.querySelector('.stillWrapper'); if(getComputedStyle(stillWrapper).display === 'block'){ stillWrapper.style.display = 'none'; stillArrow.style.transform = "rotate(180deg)"; }else{ stillWrapper.style.display = 'block'; stillArrow.style.transform = "rotate(360deg)"; } });
